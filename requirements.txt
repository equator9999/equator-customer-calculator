#!/usr/bin/env python3
"""
run_pricing.py

Adds:
- Carrier filter (UPS/DHL/FEDEX) using ZONES col B (carrier)
- Service type filter using each service tab A1 (EXPRESS/ECONOMY/DOMESTIC/FREIGHT)
- Toggle to include duties/taxes (Easyship) and optional "only for cheapest N"
- Carrier normalisation (UPS/DHL/FEDEX)
- Fuel overrides (Option A): passed from Streamlit as JSON mapping "CARRIER|TYPE" -> decimal pct

NEW:
- Transit times support (TRANSIT_BASE.xlsx + TRANSIT_CITY_OVERRIDES.xlsx + CITYCLASS_MASTER.xlsx)
  - City normalization (remove accents) + fuzzy matching for near spellings.
  - Outputs TransitDays_Min/Max, CityClass, TransitSource

- HS risk flags from hs_codes.csv (comma or pipe delimited)
  - Outputs HSCode_Used, HS_Description, HS_Risk, HS_RiskColour

- IsLowestRatePerShipment flag in output
"""

def norm(x: Any) -> str:
    return str(x).strip().upper() if x is not None else ""


def strip_accents(s: str) -> str:
    s = unicodedata.normalize("NFKD", s)
    return "".join(ch for ch in s if not unicodedata.combining(ch))


def norm_city(x: Any) -> str:
    if x is None:
        return ""
    s = str(x).strip()
    if not s:
        return ""
    s = strip_accents(s)
    s = s.upper()
    s = re.sub(r"[^A-Z0-9\s\-']", " ", s)
    s = re.sub(r"\s+", " ", s).strip()
    return s


def norm_carrier(x: Any) -> str:
    """Normalise carrier names to UPS/DHL/FEDEX."""
    s = norm(x)
    if not s:
        return ""
    if "FEDEX" in s or "FED EX" in s:
        return "FEDEX"
    if "DHL" in s:
        return "DHL"
    if "UPS" in s:
        return "UPS"
    return s


def to_float(x: Any) -> Optional[float]:
    if x is None:
        return None
    if isinstance(x, str) and x.strip() == "":
        return None
    try:
        return float(x)
    except Exception:
        try:
            return float(str(x).strip())
        except Exception:
            return None


def clean_pad_hs(hs_raw: Any) -> Optional[str]:
    """
    Clean/pad HS codes for Easyship.
    - keep digits only
    - pad to 8 if 6 or 7 digits
    - allow 8 or 10 digits (keep up to 10)
    """
    if hs_raw is None:
        return None
    s = str(hs_raw).strip()
    if not s:
        return None

    digits = re.sub(r"\D+", "", s)
    if not digits:
        return None

    if len(digits) == 6:
        return digits + "00"
    if len(digits) == 7:
        return digits + "0"
    if len(digits) == 8:
        return digits
    if len(digits) >= 10:
        return digits[:10]
    return digits


def read_table_sheet(wb: openpyxl.Workbook, name: str, expected_cols: List[str]) -> Optional[pd.DataFrame]:
    if name not in wb.sheetnames:
        return None
    ws = wb[name]
    headers = [ws.cell(1, c).value for c in range(1, ws.max_column + 1)]
    idx = {str(h).strip(): i + 1 for i, h in enumerate(headers) if h is not None and str(h).strip() != ""}
    if not all(col in idx for col in expected_cols):
        return None

    rows = []
    for r in range(2, ws.max_row + 1):
        rec = {}
        empty = True
        for col in expected_cols:
            v = ws.cell(r, idx[col]).value
            if v is not None and str(v).strip() != "":
                empty = False
            rec[col] = v
        if not empty:
            rows.append(rec)

    return pd.DataFrame(rows)


def scan_services_from_zones(wb: openpyxl.Workbook, zones_sheet: str = "ZONES") -> List[str]:
    ws = wb[zones_sheet]
    services: List[str] = []
    c = 4  # D
    while True:
        h = ws.cell(1, c).value
        if h is None or str(h).strip() == "":
            break
        services.append(str(h).strip())
        c += 1
    return services


def parse_zones(wb: openpyxl.Workbook, services: List[str]) -> Dict[str, List[dict]]:
    ws = wb["ZONES"]
    by_country: Dict[str, List[dict]] = {}

    for r in range(2, ws.max_row + 1):
        country = ws.cell(r, 3).value
        if country is None or str(country).strip() == "":
            continue

        supplier = ws.cell(r, 1).value
        carrier = ws.cell(r, 2).value
        carrier_key = norm_carrier(carrier)

        zones_by_service = {}
        for j, svc in enumerate(services):
            v = ws.cell(r, 4 + j).value
            if v is None or str(v).strip() == "":
                continue
            zones_by_service[svc] = str(v).strip()

        rec = {
            "supplier": supplier,
            "carrier": carrier,
            "carrier_key": carrier_key,
            "country": country,
            "country_key": norm(country),
            "zones_by_service": zones_by_service,
        }
        by_country.setdefault(rec["country_key"], []).append(rec)

    return by_country


def parse_service_tabs(wb: openpyxl.Workbook, services: List[str]) -> Dict[str, dict]:
    out: Dict[str, dict] = {}
    for svc in services:
        if svc not in wb.sheetnames:
            continue
        ws = wb[svc]
        mode = norm(ws["A1"].value)  # EXPRESS/ECONOMY/DOMESTIC/FREIGHT

        zone_headers: List[str] = []
        col = 2
        while True:
            v = ws.cell(1, col).value
            if v is None or str(v).strip() == "":
                break
            zone_headers.append(str(v).strip())
            col += 1

        breaks: List[float] = []
        row = 2
        while True:
            v = ws.cell(row, 1).value
            if v is None or str(v).strip() == "":
                break
            try:
                breaks.append(float(v))
            except Exception:
                try:
                    breaks.append(float(str(v).strip()))
                except Exception:
                    break
            row += 1

        prices = np.empty((len(breaks), len(zone_headers)), dtype=float)
        prices[:] = np.nan
        for i in range(len(breaks)):
            for j in range(len(zone_headers)):
                cv = ws.cell(2 + i, 2 + j).value
                if cv is None or str(cv).strip() == "":
                    continue
                try:
                    prices[i, j] = float(cv)
                except Exception:
                    try:
                        prices[i, j] = float(str(cv).strip())
                    except Exception:
                        pass

        out[svc] = {"mode": mode, "zones": zone_headers, "breaks": breaks, "prices": prices}

    return out


def next_break(breaks: List[float], w: float) -> Optional[float]:
    for b in breaks:
        if b >= w - 1e-9:
            return b
    return None


def lookup_base(service_struct: dict, service: str, zone: str, chargeable: float) -> Tuple[Optional[float], Optional[str], Optional[float], str]:
    if service not in service_struct:
        return None, None, None, "MISSING_SERVICE_TAB"

    ss = service_struct[service]
    billed = next_break(ss["breaks"], chargeable)
    if billed is None:
        return None, None, None, "NO_WEIGHT_BREAK"

    zone_str = str(zone).strip()
    if zone_str not in ss["zones"]:
        zmap = {z.upper(): z for z in ss["zones"]}
        if zone_str.upper() in zmap:
            zone_str = zmap[zone_str.upper()]
        else:
            return billed, None, None, "ZONE_NOT_IN_TARIFF"

    ridx = ss["breaks"].index(billed)
    cidx = ss["zones"].index(zone_str)
    price = ss["prices"][ridx, cidx]
    if np.isnan(price):
        return billed, zone_str, None, "PRICE_MISSING"

    return billed, zone_str, float(price), "OK"


def calc_parcels(row: dict, divisor: float, max_parcels: int = 5) -> Optional[dict]:
    parcels = []
    for i in range(1, max_parcels + 1):
        dead = to_float(row.get(f"P{i}_DeadKg"))
        L = to_float(row.get(f"P{i}_L"))
        W = to_float(row.get(f"P{i}_W"))
        H = to_float(row.get(f"P{i}_H"))

        if dead is None and L is None and W is None and H is None:
            continue

        vol = (L * W * H) / divisor if (L is not None and W is not None and H is not None) else None
        candidates = [x for x in (dead, vol) if x is not None]
        chg = max(candidates) if candidates else None

        dims = [d for d in (L, W, H) if d is not None]
        longest = max(dims) if dims else None
        len_plus_girth = None
        if len(dims) == 3:
            s = sorted(dims)
            len_plus_girth = s[2] + 2 * (s[1] + s[0])

        parcels.append(
            {
                "dead": dead,
                "L": L,
                "W": W,
                "H": H,
                "vol": vol,
                "chg": chg,
                "longest": longest,
                "len_plus_girth": len_plus_girth,
            }
        )

    if not parcels:
        return None

    pieces = len(parcels)
    chargeable = sum(p["chg"] for p in parcels if p["chg"] is not None)
    return {"parcels": parcels, "pieces": pieces, "chargeable": chargeable}


def build_maps(wb: openpyxl.Workbook):
    # Fuel map: (CarrierKey, Type) -> pct (decimal)
    fuel_map = {}
    fuel_df = read_table_sheet(wb, "FUEL", ["Carrier", "Type", "FuelPct"])
    if fuel_df is not None:
        for _, r in fuel_df.iterrows():
            try:
                pct = float(r["FuelPct"])
                if pct > 1:
                    pct /= 100.0
                fuel_map[(norm_carrier(r["Carrier"]), norm(r["Type"]))] = pct
            except Exception:
                pass

    oversize_map = {}
    os_df = read_table_sheet(wb, "OVERSIZE", ["Carrier", "LongestSideCm_Over", "FeePerPiece"])
    if os_df is not None:
        for _, r in os_df.iterrows():
            try:
                oversize_map[norm_carrier(r["Carrier"])] = (float(r["LongestSideCm_Over"]), float(r["FeePerPiece"]))
            except Exception:
                pass

    nonconv_bands = {}
    nc_df = read_table_sheet(wb, "NONCONVEYABLE", ["Carrier", "DeadKg_From", "DeadKg_To", "FeePerPiece"])
    if nc_df is not None:
        for _, r in nc_df.iterrows():
            try:
                nonconv_bands.setdefault(norm_carrier(r["Carrier"]), []).append(
                    (float(r["DeadKg_From"]), float(r["DeadKg_To"]), float(r["FeePerPiece"]))
                )
            except Exception:
                pass
        for k in nonconv_bands:
            nonconv_bands[k].sort(key=lambda t: (t[1], t[0]))

    overgirth_map = {}
    og_df = read_table_sheet(wb, "OVERGIRTH", ["Carrier", "LplusGirthCm_Over", "FeePerPiece"])
    if og_df is not None:
        for _, r in og_df.iterrows():
            try:
                overgirth_map[norm_carrier(r["Carrier"])] = (float(r["LplusGirthCm_Over"]), float(r["FeePerPiece"]))
            except Exception:
                pass

    min_map = {}
    min_df = read_table_sheet(wb, "MINIMUMS", ["Carrier", "Service", "Zone", "MinPricePerPiece"])
    if min_df is not None:
        for _, r in min_df.iterrows():
            try:
                min_map[(norm_carrier(r["Carrier"]), str(r["Service"]).strip(), str(r["Zone"]).strip())] = float(r["MinPricePerPiece"])
            except Exception:
                pass

    return fuel_map, oversize_map, nonconv_bands, overgirth_map, min_map


def calc_oversize(oversize_map, carrier_key: str, parcels: List[dict]) -> Tuple[int, float, Optional[float], Optional[float]]:
    if carrier_key not in oversize_map:
        return 0, 0.0, None, None
    thr, fee = oversize_map[carrier_key]
    pcs = sum(1 for p in parcels if p.get("longest") is not None and p["longest"] > thr)
    return pcs, pcs * fee, thr, fee


def calc_nonconv(nonconv_bands, carrier_key: str, parcels: List[dict]) -> Tuple[int, float]:
    bands = nonconv_bands.get(carrier_key)
    if not bands:
        return 0, 0.0

    pcs = 0
    amt = 0.0
    for p in parcels:
        dead = p.get("dead")
        if dead is None:
            continue
        fee = 0.0
        for a, b, f in bands:
            if dead > a and dead <= b + 1e-9:
                fee = f
                break
        if fee > 0:
            pcs += 1
            amt += fee
    return pcs, amt


def calc_overgirth(overgirth_map, carrier_key: str, parcels: List[dict]) -> Tuple[int, float, Optional[float], Optional[float]]:
    if carrier_key not in overgirth_map:
        return 0, 0.0, None, None
    thr, fee = overgirth_map[carrier_key]
    pcs = sum(1 for p in parcels if p.get("len_plus_girth") is not None and p["len_plus_girth"] > thr)
    return pcs, pcs * fee, thr, fee


def excel_safe_df(df: pd.DataFrame) -> pd.DataFrame:
    """
    Convert DataFrame values into types openpyxl can write:
    - pd.NA / NaN -> None
    - numpy scalars -> Python scalars
    - dict/list/tuple/set -> JSON string
    """
    if df is None or df.empty:
        return df

    def conv(v):
        if v is None:
            return None
        try:
            if pd.isna(v):
                return None
        except Exception:
            pass

        if isinstance(v, np.generic):
            try:
                return v.item()
            except Exception:
                return str(v)

        if isinstance(v, (dict, list, tuple, set)):
            try:
                return json.dumps(v, default=str, ensure_ascii=False)
            except Exception:
                return str(v)

        return v

    out = df.copy().astype("object")
    # applymap is fine here (we need per-cell conversion)
    return out.applymap(conv)

def country_to_alpha2(x: Any) -> Optional[str]:
    """Accept ISO2 already, or try to look up a country name."""
    if x is None:
        return None
    s = str(x).strip()
    if not s:
        return None
    if len(s) == 2 and s.isalpha():
        return s.upper()
    try:
        c = pycountry.countries.lookup(s)
        return c.alpha_2.upper()
    except Exception:
        return None


def easyship_taxes_and_duties(
    token: str,
    origin_alpha2: str,
    dest_alpha2: str,
    shipment_charge: float,
    items: List[dict],
    insurance_fee: float = 0.0,
    timeout_s: int = 30,
) -> Tuple[Optional[float], Optional[float], Optional[str]]:
    """
    Returns (duty_total, tax_total, error_message)
    Supports response shape:
      {"tax_and_duty":{"currency":"GBP","duty":0,"tax":35.11},"meta":{...}}
    """
    url = "https://public-api.easyship.com/2024-09/taxes_and_duties"
    headers = {"Authorization": f"Bearer {token}", "Content-Type": "application/json"}

    payload = {
        "origin_country_alpha2": origin_alpha2,
        "destination_country_alpha2": dest_alpha2,
        "insurance_fee": float(insurance_fee),
        "shipment_charge": float(shipment_charge),
        "items": items,
    }

    try:
        r = requests.post(url, headers=headers, json=payload, timeout=timeout_s)
        if r.status_code >= 400:
            return None, None, f"{r.status_code}: {r.text[:300]}"

        data = r.json()

        duty = None
        tax = None

        if isinstance(data, dict) and isinstance(data.get("tax_and_duty"), dict):
            tad = data["tax_and_duty"]
            if isinstance(tad.get("duty"), (int, float)):
                duty = float(tad["duty"])
            if isinstance(tad.get("tax"), (int, float)):
                tax = float(tad["tax"])

        if duty is None and tax is None:
            return None, None, f"Unrecognised response: {str(data)[:300]}"

        return duty or 0.0, tax or 0.0, None

    except Exception as e:
        return None, None, f"Request failed: {e}"

def load_transit_files(app_dir: str) -> dict:
    """
    Expects these files in repo folder:
      - TRANSIT_BASE.xlsx
      - TRANSIT_CITY_OVERRIDES.xlsx
      - CITYCLASS_MASTER.xlsx
    """
    paths = {
        "base": os.path.join(app_dir, "TRANSIT_BASE.xlsx"),
        "overrides": os.path.join(app_dir, "TRANSIT_CITY_OVERRIDES.xlsx"),
        "cityclass": os.path.join(app_dir, "CITYCLASS_MASTER.xlsx"),
    }
    out = {"ok": False, "paths": paths}

    for k, p in paths.items():
        if not os.path.exists(p):
            out["error"] = f"Missing transit file: {os.path.basename(p)}"
            return out

    base_df = pd.read_excel(paths["base"], sheet_name=0, dtype=str)
    ov_df = pd.read_excel(paths["overrides"], sheet_name=0, dtype=str)
    cc_df = pd.read_excel(paths["cityclass"], sheet_name=0, dtype=str)

    base_df.columns = [str(c).strip() for c in base_df.columns]
    ov_df.columns = [str(c).strip() for c in ov_df.columns]
    cc_df.columns = [str(c).strip() for c in cc_df.columns]

    base_map: Dict[Tuple[str, str, str, str, str], Tuple[int, int, str]] = {}
    for _, r in base_df.iterrows():
        carrier = norm_carrier(r.get("Carrier"))
        service = norm(r.get("Service"))
        origin = norm(r.get("OriginCountry"))
        region = str(r.get("DestinationRegion", "")).strip()
        cityclass = norm(r.get("CityClass"))
        tmin = int(float(str(r.get("TransitDays_Min", "0")).strip() or 0))
        tmax = int(float(str(r.get("TransitDays_Max", "0")).strip() or 0))
        notes = str(r.get("Notes", "")).strip()
        if carrier and service and origin and region and cityclass and tmin and tmax:
            base_map[(carrier, service, origin, region, cityclass)] = (tmin, tmax, notes)

    ov_map: Dict[Tuple[str, str, str, str], Tuple[str, int, int, str]] = {}
    for _, r in ov_df.iterrows():
        carrier = norm_carrier(r.get("Carrier"))
        service = norm(r.get("Service"))
        ctry = norm(r.get("DestinationCountry"))
        city = norm_city(r.get("DestinationCity"))
        cclass = norm(r.get("CityClass"))
        try:
            tmin = int(float(str(r.get("TransitDays_Min", "0")).strip() or 0))
            tmax = int(float(str(r.get("TransitDays_Max", "0")).strip() or 0))
        except Exception:
            continue
        reason = str(r.get("Reason", "")).strip()
        if carrier and service and ctry and city and tmin and tmax:
            ov_map[(carrier, service, ctry, city)] = (cclass or "", tmin, tmax, reason)

    cc_country_map: Dict[str, Dict[str, str]] = {}
    cc_country_list: Dict[str, List[str]] = {}
    for _, r in cc_df.iterrows():
        ctry = norm(r.get("Country"))
        city = norm_city(r.get("CityOrArea"))
        cclass = norm(r.get("CityClass"))
        if ctry and city and cclass:
            cc_country_map.setdefault(ctry, {})[city] = cclass

    for ctry, mp in cc_country_map.items():
        cc_country_list[ctry] = list(mp.keys())

    out.update(
        {
            "ok": True,
            "base_map": base_map,
            "ov_map": ov_map,
            "cc_country_map": cc_country_map,
            "cc_country_list": cc_country_list,
        }
    )
    return out


def country_to_region(alpha2: Optional[str]) -> str:
    if not alpha2:
        return "Asia Pacific"
    a2 = alpha2.upper()

    north_america = {"US", "CA", "MX"}
    middle_east = {"AE", "SA", "QA", "KW", "BH", "OM", "YE", "JO", "LB", "IL", "PS", "IQ", "IR", "SY", "TR"}
    europe = {
        "GB", "IE", "FR", "DE", "ES", "PT", "IT", "NL", "BE", "LU", "CH", "AT", "DK", "SE", "NO", "FI", "IS",
        "PL", "CZ", "SK", "HU", "RO", "BG", "GR", "CY", "MT", "SI", "HR", "BA", "RS", "ME", "AL", "MK",
        "EE", "LV", "LT", "UA", "MD"
    }
    africa = {
        "DZ", "AO", "BJ", "BW", "BF", "BI", "CV", "CM", "CF", "TD", "KM", "CG", "CD", "CI", "DJ", "EG", "GQ", "ER", "SZ",
        "ET", "GA", "GM", "GH", "GN", "GW", "KE", "LS", "LR", "LY", "MG", "MW", "ML", "MR", "MU", "MA", "MZ", "NA", "NE",
        "NG", "RW", "ST", "SN", "SC", "SL", "SO", "ZA", "SS", "SD", "TZ", "TG", "TN", "UG", "ZM", "ZW"
    }

    if a2 in europe:
        return "Europe"
    if a2 in north_america:
        return "North America"
    if a2 in middle_east:
        return "Middle East"
    if a2 in africa:
        return "Africa"
    return "Asia Pacific"


def classify_city(
    cc_country_map: Dict[str, Dict[str, str]],
    cc_country_list: Dict[str, List[str]],
    dest_country: str,
    dest_city_raw: Any,
) -> Tuple[str, str]:
    ctry = norm(dest_country)
    city_normed = norm_city(dest_city_raw)
    if not ctry:
        return "SECONDARY", "MISSING_COUNTRY"
    if not city_normed:
        return "SECONDARY", "DEFAULT_SECONDARY"

    mp = cc_country_map.get(ctry)
    lst = cc_country_list.get(ctry)
    if not mp or not lst:
        return "SECONDARY", "MISSING_COUNTRY"

    if city_normed in mp:
        return mp[city_normed], "EXACT"

    matches = get_close_matches(city_normed, lst, n=1, cutoff=0.88)
    if matches:
        return mp[matches[0]], "FUZZY"

    return "SECONDARY", "DEFAULT_SECONDARY"


def load_hs_db(app_dir: str) -> dict:
    path = os.path.join(app_dir, "hs_codes.csv")
    if not os.path.exists(path):
        return {"ok": False, "error": "hs_codes.csv not found in repo root", "path": path}

    df = None
    for sep in ("|", ",", "\t", ";"):
        try:
            tmp = pd.read_csv(path, sep=sep, dtype=str)
            if tmp is not None and tmp.shape[1] >= 2:
                df = tmp
                break
        except Exception:
            continue

    if df is None:
        return {"ok": False, "error": "Could not parse hs_codes.csv", "path": path}

    df.columns = [str(c).strip().lower() for c in df.columns]
    code_col = None
    desc_col = None
    risk_col = None
    for c in df.columns:
        if c in ("hscode", "hs_code", "hs"):
            code_col = c
        if c in ("description", "desc", "goods", "item"):
            desc_col = c
        if c in ("risk", "risklevel", "risk_level"):
            risk_col = c

    if code_col is None:
        code_col = df.columns[0]
    if desc_col is None and len(df.columns) >= 2:
        desc_col = df.columns[1]

    hs_map = {}
    for _, r in df.iterrows():
        code = clean_pad_hs(r.get(code_col))
        if not code:
            continue
        desc = str(r.get(desc_col, "")).strip() if desc_col else ""
        risk = norm(r.get(risk_col)) if risk_col else ""
        if risk not in ("LOW", "MED", "MEDIUM", "HIGH"):
            risk = "MED"
        if risk == "MEDIUM":
            risk = "MED"
        hs_map[code] = {"desc": desc, "risk": risk}

    return {"ok": True, "path": path, "hs_map": hs_map}


def risk_colour(risk: str) -> str:
    r = norm(risk)
    if r == "LOW":
        return "GREEN"
    if r in ("MED", "MEDIUM"):
        return "AMBER"
    if r == "HIGH":
        return "RED"
    return "GREY"


# ---------------------------
# Main
# ---------------------------
def main():
    ap = argparse.ArgumentParser()
    ap.add_argument("--rates", required=True, help="Path to rates workbook (xlsx)")
    ap.add_argument("--shipments", required=True, help="Path to shipment workbook (xlsx/xlsm)")
    ap.add_argument("--out", required=False, help="Output xlsx path.")
    ap.add_argument("--divisor", type=float, default=5000.0, help="Volumetric divisor (cm). Default 5000")
    ap.add_argument("--max_parcels", type=int, default=5, help="Max parcels per shipment row. Default 5")

    ap.add_argument("--carriers", default="", help="Comma-separated carriers to include (UPS,DHL,FEDEX). Blank=all")
    ap.add_argument("--types", default="", help="Comma-separated service types to include (EXPRESS,ECONOMY,DOMESTIC,FREIGHT). Blank=all")

    ap.add_argument("--include_customs", default="0", help="1=call Easyship, 0=skip")
    ap.add_argument("--customs_only_cheapest", default="0", help="1=only call Easyship for cheapest N per shipment")
    ap.add_argument("--customs_top_n", type=int, default=5, help="Cheapest N per shipment for customs calls")

    ap.add_argument("--fuel_overrides_json", default="", help='JSON like {"UPS|EXPRESS":0.18,"DHL|ECONOMY":0.12}')

    args = ap.parse_args()

    rates_path = args.rates
    ship_path = args.shipments
    out_path = args.out or os.path.join(os.path.dirname(os.path.abspath(ship_path)), "pricing_output.xlsx")

    include_customs = str(args.include_customs).strip() == "1"
    customs_only_cheapest = str(args.customs_only_cheapest).strip() == "1"
    customs_top_n = int(args.customs_top_n)

    carriers_selected = {norm_carrier(x) for x in str(args.carriers).split(",") if x.strip()}  # empty => all
    types_selected = {norm(x) for x in str(args.types).split(",") if x.strip()}  # empty => all

    # Parse fuel overrides
    fuel_overrides: Dict[Tuple[str, str], float] = {}
    if args.fuel_overrides_json:
        try:
            raw = json.loads(args.fuel_overrides_json)
            if isinstance(raw, dict):
                for k, v in raw.items():
                    if not isinstance(k, str) or "|" not in k:
                        continue
                    carrier, ftype = k.split("|", 1)
                    carrier = norm_carrier(carrier)
                    ftype = norm(ftype)
                    pct = float(v)
                    if pct > 1:
                        pct = pct / 100.0
                    if carrier and ftype and pct >= 0:
                        fuel_overrides[(carrier, ftype)] = pct
        except Exception:
            fuel_overrides = {}

    easyship_token = os.environ.get("EASYSHIP_ACCESS_TOKEN", "").strip()
    easyship_cache = {}

    # Load transit + HS
    app_dir = os.path.dirname(os.path.abspath(__file__))
    transit = load_transit_files(app_dir)
    hsdb = load_hs_db(app_dir)

    rates_wb = openpyxl.load_workbook(rates_path, data_only=True)
    services = scan_services_from_zones(rates_wb, "ZONES")
    zones_by_country = parse_zones(rates_wb, services)
    service_struct = parse_service_tabs(rates_wb, services)
    fuel_map, oversize_map, nonconv_bands, overgirth_map, min_map = build_maps(rates_wb)

    service_type_by_name = {svc: norm(service_struct.get(svc, {}).get("mode", "")) for svc in services}

    ship_wb = openpyxl.load_workbook(ship_path, data_only=True)
    ship_ws = ship_wb.active
    headers = [ship_ws.cell(1, c).value for c in range(1, ship_ws.max_column + 1)]

    ship_rows = []
    for r in range(2, ship_ws.max_row + 1):
        rec = {headers[c - 1]: ship_ws.cell(r, c).value for c in range(1, ship_ws.max_column + 1)}
        if rec.get("Shipment ID") is None or str(rec.get("Shipment ID")).strip() == "":
            continue
        ship_rows.append(rec)
    ship_df = pd.DataFrame(ship_rows)

    # Shipments_Calc
    calc_rows = []
    parcels_by_ship = {}
    for _, row in ship_df.iterrows():
        sid = str(row.get("Shipment ID"))
        pc = calc_parcels(row.to_dict(), args.divisor, args.max_parcels)
        parcels_by_ship[sid] = pc

        out = {"Shipment ID": sid, "To Country": row.get("To Country"), "To City": row.get("To City")}
        if pc is None:
            out["Pieces"] = 0
            out["ChargeableKg"] = np.nan
        else:
            out["Pieces"] = pc["pieces"]
            out["ChargeableKg"] = pc["chargeable"]
            for i, p in enumerate(pc["parcels"], start=1):
                out[f"P{i}_VolKg"] = p["vol"]
                out[f"P{i}_ChgKg"] = p["chg"]
                out[f"P{i}_LongestCm"] = p["longest"]
                out[f"P{i}_LenPlusGirthCm"] = p["len_plus_girth"]
        calc_rows.append(out)

    calc_df = pd.DataFrame(calc_rows)

    # Prices
    price_rows = []
    for _, row in ship_df.iterrows():
        sid = str(row.get("Shipment ID"))
        country = row.get("To Country")
        to_city = row.get("To City")
        ckey = norm(country)

        pc = parcels_by_ship.get(sid)
        if pc is None:
            price_rows.append({"Shipment ID": sid, "Country": country, "ToCity": to_city, "Status": "NO_PARCELS"})
            continue

        zlist = zones_by_country.get(ckey)
        if not zlist:
            price_rows.append({"Shipment ID": sid, "Country": country, "ToCity": to_city, "Status": "COUNTRY_NOT_FOUND"})
            continue

        pieces = pc["pieces"]
        chargeable = float(pc["chargeable"])

        # HS used (choose first parcel HS if present)
        hs_used = None
        for i in range(1, args.max_parcels + 1):
            hs_used = clean_pad_hs(row.get(f"P{i}_HSCode"))
            if hs_used:
                break
        hs_desc = ""
        hs_risk = "MED"
        hs_colour = risk_colour(hs_risk)
        if hsdb.get("ok") and hs_used and hs_used in hsdb["hs_map"]:
            hs_desc = hsdb["hs_map"][hs_used]["desc"]
            hs_risk = hsdb["hs_map"][hs_used]["risk"]
            hs_colour = risk_colour(hs_risk)

        # Transit
        dest_a2 = country_to_alpha2(country)
        region = country_to_region(dest_a2)
        cityclass = "SECONDARY"
        city_match = "DEFAULT_SECONDARY"
        if transit.get("ok"):
            cityclass, city_match = classify_city(transit["cc_country_map"], transit["cc_country_list"], dest_a2 or "", to_city)

        for zr in zlist:
            supplier = zr["supplier"]
            carrier = zr["carrier"]
            carrier_key = zr["carrier_key"]

            if carriers_selected and carrier_key not in carriers_selected:
                continue

            ov_pcs, ov_amt, ov_thr, ov_fee = calc_oversize(oversize_map, carrier_key, pc["parcels"])
            nc_pcs, nc_amt = calc_nonconv(nonconv_bands, carrier_key, pc["parcels"])
            og_pcs, og_amt, og_thr, og_fee = calc_overgirth(overgirth_map, carrier_key, pc["parcels"])

            for service, zone in zr["zones_by_service"].items():
                service_type = service_type_by_name.get(service, "")
                if types_selected and service_type and service_type not in types_selected:
                    continue
                if types_selected and not service_type:
                    continue

                billed_break, zone_used, base, base_status = lookup_base(service_struct, service, zone, chargeable)
                mode = service_struct.get(service, {}).get("mode", "")

                if base_status != "OK":
                    price_rows.append(
                        {
                            "Shipment ID": sid,
                            "Country": country,
                            "ToCity": to_city,
                            "Supplier": supplier,
                            "Carrier": carrier,
                            "CarrierKey": carrier_key,
                            "Service": service,
                            "Type": service_type,
                            "Mode": mode,
                            "Zone": zone,
                            "BilledWeightBreak": billed_break,
                            "Pieces": pieces,
                            "ChargeableKg": chargeable,
                            "BaseFreight": base,
                            "Status": base_status,
                            "HSCode_Used": hs_used,
                            "HS_Description": hs_desc,
                            "HS_Risk": hs_risk,
                            "HS_RiskColour": hs_colour,
                        }
                    )
                    continue

                subtotal_nonfuel = base + (ov_amt if ov_pcs else 0.0) + (nc_amt if nc_pcs else 0.0) + (og_amt if og_pcs else 0.0)

                # UPS minimum BEFORE fuel
                min_per_piece = min_map.get((carrier_key, service, str(zone).strip()), None)
                min_total = (min_per_piece * pieces) if min_per_piece is not None else None
                subtotal_before_fuel = subtotal_nonfuel
                min_applied = False
                if carrier_key == "UPS" and min_total is not None and subtotal_before_fuel < min_total - 1e-9:
                    subtotal_before_fuel = min_total
                    min_applied = True

                mode_key = norm(mode)
                fuel_pct = None
                fuel_source = "MISSING"
                if (carrier_key, mode_key) in fuel_overrides:
                    fuel_pct = fuel_overrides[(carrier_key, mode_key)]
                    fuel_source = "OVERRIDE"
                else:
                    fuel_pct = fuel_map.get((carrier_key, mode_key), None)
                    fuel_source = "WORKBOOK" if fuel_pct is not None else "MISSING"

                fuel_amt = (subtotal_before_fuel * fuel_pct) if fuel_pct is not None else None
                final_total = (subtotal_before_fuel + fuel_amt) if fuel_amt is not None else subtotal_before_fuel

                # Transit lookup
                tmin = None
                tmax = None
                tsource = "MISSING"
                tnotes = ""
                if transit.get("ok"):
                    city_normed = norm_city(to_city)
                    if dest_a2 and city_normed and (carrier_key, norm(mode_key), dest_a2, city_normed) in transit["ov_map"]:
                        cclass_ov, omin, omax, reason = transit["ov_map"][(carrier_key, norm(mode_key), dest_a2, city_normed)]
                        tmin, tmax = omin, omax
                        tsource = "CITY_OVERRIDE"
                        tnotes = reason
                        if cclass_ov:
                            cityclass = cclass_ov
                    else:
                        key = (carrier_key, norm(mode_key), "GB", region, cityclass)
                        if key in transit["base_map"]:
                            tmin, tmax, tnotes = transit["base_map"][key]
                            tsource = "BASE"
                        else:
                            for cc_try in ("METRO", "SECONDARY", "REMOTE"):
                                key2 = (carrier_key, norm(mode_key), "GB", region, cc_try)
                                if key2 in transit["base_map"]:
                                    tmin, tmax, tnotes = transit["base_map"][key2]
                                    tsource = f"BASE_FALLBACK_{cc_try}"
                                    break

                # Customs inputs
                origin_raw = row.get("Origin Country")
                currency = row.get("Currency")
                incoterm = row.get("Incoterm")
                origin_a2 = country_to_alpha2(origin_raw)

                items = []
                for i in range(1, args.max_parcels + 1):
                    hs_clean = clean_pad_hs(row.get(f"P{i}_HSCode"))
                    val = to_float(row.get(f"P{i}_Value"))
                    if not hs_clean:
                        continue
                    if val is None or val <= 0:
                        continue
                    if origin_a2 is None:
                        continue
                    items.append(
                        {
                            "duty_origin_country_alpha2": origin_a2,
                            "hs_code": hs_clean,
                            "customs_value": float(val),
                        }
                    )

                goods_value_total = float(sum(it["customs_value"] for it in items)) if items else None

                price_rows.append(
                    {
                        "Shipment ID": sid,
                        "Country": country,
                        "ToCity": to_city,
                        "Supplier": supplier,
                        "Carrier": carrier,
                        "CarrierKey": carrier_key,
                        "Service": service,
                        "Type": service_type,
                        "Mode": mode,
                        "Zone": zone,
                        "BilledWeightBreak": billed_break,
                        "Pieces": pieces,
                        "ChargeableKg": chargeable,
                        "BaseFreight": base,
                        "Status": "OK",
                        "OversizeThresholdCm": ov_thr,
                        "OversizePieces": ov_pcs,
                        "OversizeFeePerPiece": ov_fee,
                        "OversizeAmount": ov_amt if ov_pcs else 0.0,
                        "NonConvPieces": nc_pcs,
                        "NonConvAmount": nc_amt if nc_pcs else 0.0,
                        "OverGirthThresholdCm": og_thr,
                        "OverGirthPieces": og_pcs,
                        "OverGirthFeePerPiece": og_fee,
                        "OverGirthAmount": og_amt if og_pcs else 0.0,
                        "SubtotalNonFuel": subtotal_nonfuel,
                        "UPS_MinPerPiece": min_per_piece,
                        "UPS_MinTotal": min_total,
                        "UPS_MinApplied": min_applied,
                        "SubtotalBeforeFuel": subtotal_before_fuel,
                        "FuelPct": fuel_pct,
                        "FuelPctSource": fuel_source,
                        "FuelAmount": fuel_amt,
                        "FinalTotal": final_total,
                        # Transit outputs
                        "DestCountryISO2": dest_a2,
                        "DestRegion": region,
                        "CityClass": cityclass,
                        "CityMatchMethod": city_match,
                        "TransitDays_Min": tmin,
                        "TransitDays_Max": tmax,
                        "TransitSource": tsource,
                        "TransitNotes": tnotes,
                        # HS outputs
                        "HSCode_Used": hs_used,
                        "HS_Description": hs_desc,
                        "HS_Risk": hs_risk,
                        "HS_RiskColour": hs_colour,
                        # Customs outputs
                        "OriginCountry": origin_a2,
                        "DestCountry": dest_a2,
                        "Incoterm": str(incoterm).strip().upper() if incoterm else None,
                        "Currency": str(currency).strip().upper() if currency else None,
                        "GoodsValueTotal": goods_value_total,
                        "DutyTotal": None,
                        "TaxTotal": None,
                        "LandedCostTotal": None,
                        "CustomsStatus": "SKIPPED" if not include_customs else "PENDING",
                        "_has_items": bool(items),
                        "_items": items,
                    }
                )

    prices_df = pd.DataFrame(price_rows)

    # ---------------------------------------------------------------------
    # Schema hardening: prevent KeyError when df is empty or columns absent
    # ---------------------------------------------------------------------
    prices_df.columns = [str(c).strip() for c in prices_df.columns]
    rename_map = {
        "Final Total": "FinalTotal",
        "Final_Total": "FinalTotal",
        "finaltotal": "FinalTotal",
        "ShipmentID": "Shipment ID",
        "Shipment Id": "Shipment ID",
    }
    prices_df = prices_df.rename(columns={k: v for k, v in rename_map.items() if k in prices_df.columns})

    must_exist = ["Shipment ID", "Status", "FinalTotal", "CustomsStatus", "OriginCountry", "DestCountry", "_has_items", "_items"]
    for c in must_exist:
        if c not in prices_df.columns:
            prices_df[c] = pd.NA

    # ---------------------------
    # Customs stage
    # ---------------------------
    if include_customs:
        if not easyship_token:
            prices_df.loc[prices_df["CustomsStatus"] == "PENDING", "CustomsStatus"] = "SKIPPED_NO_TOKEN"
        else:
            candidates = prices_df[
                (prices_df["Status"].eq("OK")) & (prices_df["CustomsStatus"].eq("PENDING"))
            ].copy()

            if not candidates.empty:
                candidates = candidates[
                    candidates["_has_items"].fillna(False)
                    & candidates["OriginCountry"].notna()
                    & candidates["DestCountry"].notna()
                    & candidates["FinalTotal"].notna()
                ].copy()

            if candidates.empty:
                prices_df.loc[prices_df["CustomsStatus"] == "PENDING", "CustomsStatus"] = "SKIPPED_NO_HS_VALUE"
            else:
                if customs_only_cheapest:
                    call_idx = []
                    for sid, grp in candidates.groupby("Shipment ID"):
                        grp2 = grp.sort_values("FinalTotal").head(customs_top_n)
                        call_idx.extend(list(grp2.index))
                    candidates = candidates.loc[call_idx]

                    mask_other_pending = (prices_df["CustomsStatus"] == "PENDING") & (~prices_df.index.isin(candidates.index))
                    prices_df.loc[mask_other_pending, "CustomsStatus"] = "SKIPPED_NOT_IN_CHEAPEST"

                for idx, r in candidates.iterrows():
                    origin_a2 = r["OriginCountry"]
                    dest_a2 = r["DestCountry"]
                    final_total = float(r["FinalTotal"])
                    items = r["_items"] if isinstance(r["_items"], list) else None

                    if not items:
                        prices_df.at[idx, "CustomsStatus"] = "SKIPPED_NO_HS_VALUE"
                        continue

                    cache_key = (
                        origin_a2,
                        dest_a2,
                        round(final_total, 2),
                        tuple((it.get("hs_code"), round(float(it.get("customs_value", 0.0)), 2)) for it in items),
                    )
                    if cache_key in easyship_cache:
                        duty_total, tax_total, customs_status = easyship_cache[cache_key]
                    else:
                        duty_total, tax_total, err = easyship_taxes_and_duties(
                            token=easyship_token,
                            origin_alpha2=origin_a2,
                            dest_alpha2=dest_a2,
                            shipment_charge=final_total,
                            items=items,
                        )
                        customs_status = "OK" if err is None else f"ERROR: {err}"
                        easyship_cache[cache_key] = (duty_total, tax_total, customs_status)

                    prices_df.at[idx, "DutyTotal"] = duty_total
                    prices_df.at[idx, "TaxTotal"] = tax_total
                    prices_df.at[idx, "CustomsStatus"] = customs_status
                    if customs_status == "OK":
                        prices_df.at[idx, "LandedCostTotal"] = float(final_total) + float(duty_total or 0.0) + float(tax_total or 0.0)

                prices_df.loc[prices_df["CustomsStatus"] == "PENDING", "CustomsStatus"] = "SKIPPED"

    # Drop internal columns (important before Excel export)
    for col in ["_has_items", "_items"]:
        if col in prices_df.columns:
            prices_df.drop(columns=[col], inplace=True)

    # IsLowestRatePerShipment (safe)
    prices_df["IsLowestRatePerShipment"] = False
    status_ok = prices_df["Status"].eq("OK")
    has_final = prices_df["FinalTotal"].notna()
    ok_mask = status_ok & has_final
    if ok_mask.any():
        ok_df = prices_df.loc[ok_mask, ["Shipment ID", "FinalTotal"]].copy()
        idxmin = ok_df.groupby("Shipment ID")["FinalTotal"].idxmin()
        prices_df.loc[idxmin, "IsLowestRatePerShipment"] = True

    # Cheapest 10 per shipment (by FinalTotal)
    ok = prices_df[(prices_df["Status"].eq("OK")) & prices_df["FinalTotal"].notna()].copy()
    cheapest = []
    if not ok.empty:
        for sid, grp in ok.groupby("Shipment ID"):
            cheapest.append(grp.sort_values("FinalTotal").head(10))
    cheap_df = pd.concat(cheapest, ignore_index=True) if cheapest else pd.DataFrame()

    # ---------------------------
    # Write output workbook
    # ---------------------------
    # Make dataframes Excel-safe (prevents "Cannot convert ... to Excel")
    calc_df_x = excel_safe_df(calc_df) if isinstance(calc_df, pd.DataFrame) else calc_df
    prices_df_x = excel_safe_df(prices_df) if isinstance(prices_df, pd.DataFrame) else prices_df
    cheap_df_x = excel_safe_df(cheap_df) if isinstance(cheap_df, pd.DataFrame) else cheap_df

    wb_out = Workbook()

    ws = wb_out.active
    ws.title = "Settings"
    ws.append(["Generated", datetime.now().isoformat(timespec="seconds")])
    ws.append(["Rates workbook", os.path.basename(rates_path)])
    ws.append(["Shipments workbook", os.path.basename(ship_path)])
    ws.append(["Volumetric divisor (cm)", args.divisor])
    ws.append(["Max parcels", args.max_parcels])
    ws.append(["Services scanned from ZONES row 1 until blank", len(services)])
    ws.append(["Carriers filter", ",".join(sorted(carriers_selected)) if carriers_selected else "ALL"])
    ws.append(["Types filter", ",".join(sorted(types_selected)) if types_selected else "ALL"])
    ws.append(["Include customs", include_customs])
    ws.append(["Customs only cheapest", customs_only_cheapest])
    ws.append(["Customs top N", customs_top_n])
    ws.append(["Easyship token present", bool(easyship_token)])
    ws.append(["Fuel overrides present", bool(fuel_overrides)])
    ws.append(["Transit files loaded", bool(transit.get("ok"))])
    ws.append(["HS DB loaded", bool(hsdb.get("ok"))])
    for cell in ws[1]:
        cell.font = Font(bold=True)

    ws2 = wb_out.create_sheet("Shipments_Calc")
    for r in dataframe_to_rows(calc_df_x, index=False, header=True):
        ws2.append(r)
    for cell in ws2[1]:
        cell.font = Font(bold=True)
    ws2.freeze_panes = "A2"

    ws3 = wb_out.create_sheet("Prices_All_Services")
    for r in dataframe_to_rows(prices_df_x, index=False, header=True):
        ws3.append(r)
    for cell in ws3[1]:
        cell.font = Font(bold=True)
    ws3.freeze_panes = "A2"

    ws4 = wb_out.create_sheet("Cheapest10_Final")
    for r in dataframe_to_rows(cheap_df_x, index=False, header=True):
        ws4.append(r)
    if ws4.max_row >= 1:
        for cell in ws4[1]:
            cell.font = Font(bold=True)
    ws4.freeze_panes = "A2"

    # Autofit
    def autofit(sheet, max_width=55):
        for col in sheet.columns:
            col_letter = col[0].column_letter
            maxlen = 0
            for cell in col:
                v = "" if cell.value is None else str(cell.value)
                maxlen = max(maxlen, len(v))
            sheet.column_dimensions[col_letter].width = min(maxlen + 2, max_width)

    for sheet in (ws, ws2, ws3, ws4):
        autofit(sheet)

    # Formatting helpers
    def format_two_decimals(sheet, column_names):
        header_row = list(sheet[1])
        headers = {cell.value: idx + 1 for idx, cell in enumerate(header_row) if cell.value is not None}
        for col_name in column_names:
            if col_name not in headers:
                continue
            col_idx = headers[col_name]
            for r in range(2, sheet.max_row + 1):
                cell = sheet.cell(row=r, column=col_idx)
                if isinstance(cell.value, (int, float)):
                    cell.number_format = "0.00"

    def format_percent(sheet, column_names):
        header_row = list(sheet[1])
        headers = {cell.value: idx + 1 for idx, cell in enumerate(header_row) if cell.value is not None}
        for col_name in column_names:
            if col_name not in headers:
                continue
            col_idx = headers[col_name]
            for r in range(2, sheet.max_row + 1):
                cell = sheet.cell(row=r, column=col_idx)
                if isinstance(cell.value, (int, float)):
                    cell.number_format = "0.00%"

    columns_2dp = [
        "ChargeableKg",
        "BaseFreight",
        "OversizeAmount",
        "NonConvAmount",
        "OverGirthAmount",
        "SubtotalNonFuel",
        "UPS_MinPerPiece",
        "UPS_MinTotal",
        "SubtotalBeforeFuel",
        "FuelAmount",
        "FinalTotal",
        "GoodsValueTotal",
        "DutyTotal",
        "TaxTotal",
        "LandedCostTotal",
    ]
    columns_pct = ["FuelPct"]

    format_two_decimals(ws3, columns_2dp)
    format_two_decimals(ws4, columns_2dp)
    format_percent(ws3, columns_pct)
    format_percent(ws4, columns_pct)

    wb_out.save(out_path)
    print(out_path)
    return


if __name__ == "__main__":
    main()
